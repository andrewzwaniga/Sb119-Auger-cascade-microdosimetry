# Andrew Zwaniga PhD Thesis Work

# Title: DNA SSB and DSB damage due to In111 source in a DNA target
# Purpose: Calculate DNA damage in the form of SSBs and DSBs due to a custom
# In111 source using G4_WATER with the TsCharltonDNA model 

# Description: Simulation of In111 decays in proximity to the TsCharltonDNA model in G4_WATER with g4em-dna_opt2
# and generic-ion model Auger electron spectrum and 1000000 histories and seed 501152503. The In111 source is placed at varying 
# distances along an axis perpendicular to the DNA helix symmetry axis. 

# Reference data for Auger electron spectrum: 
# [1] Guatelli, S., Mantero, A., Mascialino, B., Nieminen, P. & Pia, M. G. Geant4 Atomic Relaxation. IEEE Transactions on Nuclear Science 54, 585–593 (2007).

# [2] Guatelli, S., Mantero, A., Mascialino, B., Pia, M. G. & Zampichelli, V. Validation of Geant4 Atomic Relaxation Against the NIST Physical Reference Data. IEEE Transactions on Nuclear Science 54, 594–603 (2007).


# ----------- GEOMETRY ----------- #

## World
d:Ge/World/HLX = 1. um
d:Ge/World/HLY = 1. um
d:Ge/World/HLZ = 1. um
b:Ge/World/Invisible = "False"
s:Ge/World/Material = "G4_WATER"

## DNA Target
s:Ge/DNA/Type = "TsCharltonDNA"
s:Ge/DNA/Parent = "World"
s:Ge/DNA/Material = "G4_WATER"
i:Ge/DNA/NumberOfBasePairs = 10

## Remark: In order to score strand breaks, we must modify the water material

## Material for sugar backbone in strand 1
s:Ma/G4_WATER_MODIFIED_1/CloneFromMaterial = "G4_WATER"
d:Ma/G4_WATER_MODIFIED_1/CloneWithDensity = 1.0 g/cm3

## Material for sugar backbone in strand 2
s:Ma/G4_WATER_MODIFIED_2/CloneFromMaterial = "G4_WATER_MODIFIED_1"
d:Ma/G4_WATER_MODIFIED_2/CloneWithDensity = 1.0 g/cm3

## Modified water material is required to score strand breaks 
s:Ge/DNA/BasePair/Material = "G4_WATER"
s:Ge/DNA/Backbone1/Material = "G4_WATER_MODIFIED_1"
s:Ge/DNA/Backbone2/Material = "G4_WATER_MODIFIED_2"

s:Ge/DNA/BasePair/DrawingStyle = "solid"
s:Ge/DNA/Backbone1/Color = "red"
s:Ge/DNA/Backbone1/DrawingStyle = "solid"
s:Ge/DNA/Backbone2/Color = "blue"
s:Ge/DNA/Backbone2/DrawingStyle = "solid"

## Beamposition object is used to define the origin point of the source
s:Ge/Beamposition/Parent = "World"
s:Ge/Beamposition/Type = "Group"
## The source is moved along an axis perpendicular to the symmetry axis of the cylinder
d:Ge/Beamposition/TransY = 0. m
d:Ge/Beamposition/TransZ = 0. m
d:Ge/Beamposition/RotX = 0. deg
d:Ge/Beamposition/RotY = 0. deg
d:Ge/Beamposition/RotZ = 0. deg

# ----------- SOURCE ----------- #

s:So/In111/BeamParticle = "GenericIon(49,111,0)"
d:So/In111/BeamEnergy = 0. eV
s:So/In111/Type = "Isotropic"
s:So/In111/Component = "Beamposition"


# ----------- SCORING ----------- #

s:Sc/SimpleSSBandDSB/Quantity = "SimpleSSBandDSB"
s:Sc/SimpleSSBandDSB/Component = "DNA"

## Pass detector properties of the Parent to the Child geometries
b:Sc/SimpleSSBandDSB/PropagateToChildren = "True"
## In this simplified model, the sugar backbone and base pairs are replica volumes
i:Sc/SimpleSSBandDSB/BasePairPositionAtGeometricHierarchy = 0
## In order that the scorer can differentiate between strand 1 and 2 (sugar1, sugar2)
s:Sc/SimpleSSBandDSB/Strand1MaterialName = "G4_WATER_MODIFIED_1"
s:Sc/SimpleSSBandDSB/Strand2MaterialName = "G4_WATER_MODIFIED_2"
## Only record hits to the scorer's calculations if they occur in the sugar volumes
sv:Sc/SimpleSSBandDSB/OnlyIncludeIfInMaterial = 2 "G4_WATER_MODIFIED_1" "G4_WATER_MODIFIED_2"

s:Sc/SimpleSSBandDSB/OutputType = "ASCII"
s:Sc/SimpleSSBandDSB/OutputFile = "In111-SimpleSSBandDSB"
b:Sc/SimpleSSBandDSB/OutputToConsole = "True"
s:Sc/SimpleSSBandDSB/IfOutputFileAlreadyExists = "Overwrite"

# ----------- PHYSICS ----------- #

sv:Ph/Default/Modules = 2 "g4em-dna_opt2" "g4radioactivedecay"

d:Ph/Default/EMRangeMin = 10 eV
d:Ph/Default/SetProductionCutLowerEdge = 10 eV
d:Ph/Default/MaxStepSize = 0.1 nm
b:Ph/Default/Fluorescence = "True"
b:Ph/Default/Auger = "True"
b:Ph/Default/AugerCascade = "True"
b:Ph/Default/DeexcitationIgnoreCut = "True"
d:Ph/Default/DefaultRegionForTheWorld/CutForAllParticles = 0.01 nm

# ----------- KEY ADJUSTABLES ----------- #

# Number of threads to use (change to match number of cores requested)
i:Ts/NumberOfThreads = 8
b:Ts/ShowCPUTime = "True"
i:Ts/ShowHistoryCountAtInterval = 100000

# Exit if more than 0.1 percent of the total number of histories to be simulated are interrupted
i:Ts/MaxInterruptedHistories = 1000
# Stop printing out a report of interrupted histories after this many reports
i:Ts/InterruptedHistoryMaxReports = 10
# Limit the number of reports made about killed tracks
i:Ts/KilledTrackMaxReports = 10
# Exit if more than 0.1 percent of the total number of histories to be simulated have killed tracks
i:Ts/KilledTrackMaxCount = 1000

## Modeling parameters that may need adjustment
i:Sc/SimpleSSBandDSB/BasePairDistanceForDefiningDSB = 10
d:Sc/SimpleSSBandDSB/EnergyThresholdForHavingSSB = 17.5 eV

## Radial position of source relative to DNA molecule axis of symmetry 
d:Ge/Beamposition/TransX = 10. nm

## Number of histories to simulate

i:So/In111/NumberOfHistoriesInRun = 1000000


i:Ts/seed = 501152503
